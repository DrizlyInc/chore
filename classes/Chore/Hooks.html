Helper method to look up, and execute hooks based on an event name. Hooks are assumed to be methods defined on `self` that are of the pattern hook_name_identifier. ex: before_perform_log

def run_hooks_for(event,*args)
  results = global_hooks_for(event).map { |prc| prc.call(*args) } || [true]
  results << hooks_for(event).map { |method| send(method,*args) }
  results = false if results.any? {|r| false == r }
  results
end
