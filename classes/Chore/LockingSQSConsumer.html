def consume(&handler)
  while running?
    begin
      if enabled?
        if requires_lock?
          semaphore = Semaphore.new(@queue_name, @@zk)
          semaphore.acquire do
            handle_messages(&handler)
          end
        else
          handle_messages(&handler)
        end
      end
    rescue => e
      Chore.logger.error { "LockingSQSConsumer#Consume: #{e.inspect}" }
    end
  end
ensure
  @@zk.close!
end
