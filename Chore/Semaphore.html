def initialize(resource_name, zk, opts = {})
  opts = DEFAULT_OPTIONS.merge(opts)

  @zk = zk
  @resource_name = resource_name
  @path = opts[:path]
  @resource_path = "#{@path}/#{@resource_name}"
  @queue = Queue.new
  @subscription = nil

  build_path!
end

def acquire(&block)
  if block
    wait_for_lease(&block)
  else
    if available?
      Lease.new(create_lease!, @zk)
    else
      nil
    end
  end
end
